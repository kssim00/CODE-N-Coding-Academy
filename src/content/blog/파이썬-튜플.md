        const el = document.getElementById(id);

    }

    // --- 채점 로직 ---
    function gradeQuiz() {
        let score = 0;
        
        quizData.forEach((item, index) => {
            const selected = document.querySelector(`input[name="q${index}"]:checked`);
            const msgBox = document.getElementById(`msg${index}`);
            
            if (selected) {
                const val = parseInt(selected.value);
                if (val === item.a) {
                    score++;
                    msgBox.innerHTML = '<span class="correct-answer">✅ 정답입니다!</span>';
                } else {
                    msgBox.innerHTML = `<span class="wrong-answer">❌ 오답입니다.</span> 정답: ${item.options[item.a]}`;
                }
            } else {
                msgBox.innerHTML = '<span style="color:orange;">⚠️ 문제를 풀어주세요.</span>';
            }
        });

        // 12문제 기준 점수 환산
        const finalScore = Math.round((score / quizData.length) * 100);
        const resultText = document.getElementById('finalResult');
        resultText.innerText = `총점: ${finalScore}점 (${score}/${quizData.length}개)`;
        resultText.style.color = (finalScore >= 70) ? 'var(--primary)' : 'var(--error)';
    }
